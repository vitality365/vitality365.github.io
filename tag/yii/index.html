安装php

sudo apt install php7.2

安装nginx

sudo apt install nginx-full

安装对应的php-fpm

sudo apt install php7.2-fpm sudo apt install php7.2-mysql

配置网站目录

sudo vim /etc/nginx/sites-available/default
